\begin{esercizio}
	Let $G$ be any countable group. Consider $2^G, (2^\omega)^G$ equipped with the shift action.
	Prove that there exists an injective, $G$-equivariant function from $2^G$ to $(2^\omega)^G$.
\end{esercizio}

\begin{proof}
	Consider $f: 2^G \to (2^\omega)^G, \ x \mapsto x^{*}$, where
	$\forall g \in G, n < \omega \ x^{*}(g)(n) := x(g)$.
	We want to prove:
	\begin{itemize}
		\item $f$ is $G$-equivariant: \\
			Let $g \in G, x \in 2^{\omega}$. We want to prove that $f(g \cdot x) = g \cdot f(x)$.
			We evaluate both and check they are the same:
			\[
				f(g \cdot x)(h)(n) = (g \cdot x)^{*} (h)(n) =
				(g \cdot x) (h) = x(g^{-1} \cdot h)
			\]

			\[
				(g \cdot f(x))(h)(n) = (g \cdot x^{*})(h)(n) = 
				x^{*}(g^{-1} \cdot h)(n) = x(g^{-1} \cdot h)
			\]

		\item $f$ is injective: \\
			Let $x_1 \not = x_2 \in 2^{\omega}$. Then $x_1 (g) \not = x_2(g)$ for some $g \in G$.
			Then $x_1 ^{*}(g)(0) = x_1 (g) \not = x_2 (g) = x_2 ^{*}(g)(0)$.
			Thus $x_1 ^{*} \not = x_2 ^{*}$.

	\end{itemize}

\end{proof}





\begin{esercizio}
	Let $G,H$ be countable groups and $Y$ a Polish space.
	Prove that if $\{1\} \not = G \leq H$, then:
	\begin{enumerate}
		\item $E(G,Y) \leq_B E(H,Y)$
		\item $F(G,Y) \leq_B F(H,Y)$
	\end{enumerate}
\end{esercizio}

\begin{proof}
	Fix $y_0 \in Y$. Then consider $f: Y^G \to Y^H, \ p \mapsto p^{*}$, where 
	$p^{*}(h) := 
	\begin{cases}
		p(h) & \text{ if } h \in G  \\
		y_0  & else
	\end{cases}
	$

	We must prove:
	\begin{itemize}
		\item $f$ is a reduction from $E(G,Y)$ to $E(H,Y)$ \\
			\begin{itemize}
				\item $\forall p,q \in Y^{G}, g \in G \ \bigg( p = g \cdot q \Rightarrow p^{*} = g \cdot q^{*} \bigg)$ \\

					Let's consider two cases:
					\begin{itemize}
						\item $h \in G$
							\[
								p^{*} (h) = p(h) = (g \cdot q)(h) = 
								q(g^{-1} \cdot h) = q^{*}(g^{-1} \cdot h) =
								(g \cdot q^{*})(h)
							\]
						\item $h \not \in G$ 
							\[
								p^{*} (h) = y_0 = q^{*}(g^{-1} \cdot h) = (g \cdot q^{*})(h)
							\]

					\end{itemize}

				\item $\forall p,q \in Y^{G}, h \in H \ 
					\bigg( p^{*} = h \cdot q^{*} \Rightarrow (\exists g \in G \ p^{*} = g \cdot q^{*}) \bigg)$ \\

					Let $h \in H \setminus G$ such that $p^{*} = h \cdot q^{*}$.
					Then 
					\[
						\forall t \in H \ \ 
						p^{*}(t) = (h \cdot q^{*})(t) = q^{*}(h^{-1} \cdot t)
					\]
					
					In particular 
					\[
						\forall g \in G \ \ y_0 = p^{*}(h \cdot g) = q^{*}(g) = q(g)
					\]

					Then $q = cost_{y_0}$. Then $q^{*} = cost_{y_0}$. Then $p^{*} = cost_{y_0}$.
					Then $p^{*} = 1_{H} \cdot q^{*}$.


				\item $\forall p,q \in Y^{G}, g \in G 
					\bigg( p^{*} = g \cdot q^{*} \Rightarrow p = g \cdot q \bigg) $ 

					\begin{align*}
						p^{*} = g \cdot q^{*} \Rightarrow & \forall h \in H \ p^{*}(h) = q^{*}(g^{-1} \cdot h) \\
						\Rightarrow & \forall t \in G \ p(t) = p^{*}(t) = q^{*}(g^{-1} \cdot t) = 
										q(g^{-1} \cdot t) = (g \cdot q)(t)
					\end{align*}
			\end{itemize}

		\item $f$ is Borel \\
			\[
				(s,r) \in f \iff \bigg( (\forall g \in G \ s(g) = r(g)) \land (\forall h \in H \setminus G \ r(h) = y_0) \bigg)
			\]

	\end{itemize}

	For the second part, consider $f^{'}: ((Y))^G \to ((Y))^H, \ p \mapsto p^{*}$, where 
	$p^{*}(h) := 
	\begin{cases}
		p(h) & \text{ if } h \in G  \\
		y_0  & else
	\end{cases}
	$

	We claim as above that this is the desired reduction. 

	\begin{itemize}
		\item well defined: \\
			We have to check that 
			$\forall p \in ((Y))^G \ p^{*} \in ((Y))^{H}$.
			Let $h \in H$ such that $h \cdot p^{*} = p^{*}$.
			Then $\forall t \in H \ (h \cdot p^{*})(t) = p^{*}(t)$.
			Equivalently $\forall t \in H \ p^{*}(h^{-1} \cdot t) = p^{*}(t)$.
			In particular $\forall g \in G \ p^{*}(h^{-1} \cdot h \cdot g) = p^{*}(h \cdot g)$.
			Then $\forall g \in G \ p^{*}(g) = p^{*}(h \cdot g)$.
			Now there are two possibilities:
			\begin{itemize}
				\item $h \in H \setminus G$: \\
					In this case we have 
					$\forall g \in G \ p(g) = p^{*}(g) = p^{*}(h \cdot g) = y_0$.
					This means that $p = cost_{y_0}$.
					Since $G \not = \{1\}$ we have $p \not \in ((Y))^{G}$, a contradiction.
				\item $h \in G$: \\
					In this case we have 
					$\forall g \in G \ p(g) = p^{*}(g) = p^{*}(h \cdot g) = p(h \cdot g) = (h^{-1} \cdot p)(g)$
					This means that $p = h^{-1} \cdot p$.
					Since $p \in ((Y))^{G}$, it must be $h^{-1} = 1$. Then $h = 1$, as desired.
			\end{itemize}
		\item reduction: \\
			We must prove that $\forall p,q \in ((Y))^G \ \bigg( p F(G,Y) q \iff p^{*} F(H,Y) q^{*} \bigg)$
			This follows basically from $f$ being a reduction.
			In fact:
			$p F(G,Y) q \iff p E(G,Y) q \iff f(p) E(H,Y) f(q)
					\iff f'(p) E(H,Y) f'(q) \iff f'(p) F(H,Y) f'(q)$

		\item Borel: \\
			This follows from the Borelness of $f$, since
			$f' = f \cap \bigg( ((Y))^G \times ((Y))^H \bigg)$
	\end{itemize}

\end{proof}




\begin{esercizio}
	Let $E,F$ cber on $X$. Prove that if $E \subseteq F$ and $F$ is treeable, then $E$ is also treeable.
\end{esercizio}

\begin{proof}
	Let $T$ be a Borel graphing witnessing the treeability of $F$.
	Let $F = E_{G}^{X}$ for some countable group $G$. 
	Let's fix an enumeration: $G = \{g_n : n < \omega \}$.
	Call $\langle x_0,..,x_n \rangle \in X^{< \omega}$ a T-path if $\forall i<n \ \ x_i T x_{i+1}$.
	Given $x \in X$ and an $E$-class $C$, denote $PATH_{x,C} := \{s \in X^{< \omega} : s\text{ is a }T-path\text{ starting from }x 
	\text{ and ending in }C\}$.
	Now, given $x,C$ as above we can define
	$\phi: PATH_{x,C} \to \omega^{< \omega}, \langle x, y_1, .., y_k\rangle \mapsto \langle i_1,..,i_k \rangle$, where 
	$i_j = \text{min}\{n \in \omega : g_n \cdot y_{j-1} = y_j\}$ (consider $x$ as $y_0$ to evaluate $i_1$).
	In particular $\phi(x) = \emptyset$.
	One can consider the lexicographic order on $\omega ^{< \omega}$, $\leq_{lex}$, 
	where in particular we put $s \leq t$ if $s$ is an initial segment of $t$.
	Since $\phi$ is injective and $\leq_{lex}$ is a well order, we can define a well order on $PATH_{x,C}$, $\leq_{x,C}$, by setting
	\[
		\forall s,t \in PATH_{x,C} \ \big(s \leq_{x,C} t \iff \phi(s) \leq_{lex} \phi(t) \big)
	\]
	Now consider $\rho: X^2 \to X$, with 
	\[
	\rho(x,y) :=
	\begin{cases}
		\text{the last element of min}(PATH_{x,[y]_E}) & \text{ if } x F y   \\
		x & \text{else}
	\end{cases}
	\]
	Note that $\rho(x,y) = x \iff \bigg( x E y \lor x \not F y\bigg)$.
	Now we can define a Borel treeing for $E$.
	Consider $t \subseteq X^2$ given by
	\[
		x_1 t x_2 \iff 
		\bigg( x_1 E x_2 \land 
		\big(\exists i < \omega \exists j < \omega 
		(g_i \cdot x_1 T g_j \cdot x_2 \land \rho(g_i \cdot x_1,x_1) = x_1 \land \rho(g_j \cdot x_2,x_2) = x_2)
		\big) \bigg)
	\]

	We have to prove the following claims to conclude.
	\begin{itemize}
		\item $t$ is Borel \\
			This follows from the Borelness of $E,T,\rho$. we prove the Borelness of $\rho$ below.
		\item $\rho$ is Borel \\
			\[
				(x,y,z) \in \rho \iff 
				\bigg( (x = z \land x \not F y) \lor (x F y \land \text{min}(PATH_{x,[y]_E}) \text{ ends in }z )\bigg)
			\]

			Before expanding the last part it is useful introducing the following notation:
			given $\vec{l} \in \omega^{< \omega}$, define
			\[
				\langle x, g_{\vec{l}} \cdot x \rangle :=
				\begin{cases}
					\langle x \rangle & \text{ if length}(\vec{l}) = 0 \\
					\langle x, g_{\vec{l} \restriction \text{length}(\vec{l}-1)} \cdot x \rangle \frown 
					\langle (\prod_{i \leq \text{length}(\vec{l})} g_{\text{length}(\vec{l}) -i}) \cdot x \rangle & \text{ else }
				\end{cases}
			\]

			Now $\text{min}(PATH_{x,[y]_E})$ ends in $z$ can be expressed as:
			\begin{align*}
				\exists n< \omega 
				\exists \vec{i} \in \omega^{n}
				\bigg(
				& (\langle x, g_{\vec{i}} \rangle \text{ is }T \text{-path ending in }z) 
				\land \\
				& (\forall k<n \forall \vec{j} \in \omega^{k} 
					\langle x, g_{\vec{j}} \cdot x \rangle(k) \in [y]_E \to 
					\langle x, g_{\vec{j}} \cdot x \rangle \text{ is not }T\text{-path})
				\land \\
				& (\forall \vec{l} \in \omega^{n}
					\langle x, g_{\vec{j}} \cdot x \rangle(k) \in [y]_E \to
					\langle x, g_{\vec{i}} \cdot x \rangle \leq_{x,[y]_E} \langle x, g_{\vec{l}} \cdot x\rangle
				)
				\bigg)
			\end{align*}

		\item $t$ does not contain cycles \\
			By contradiction, let $\langle r_0 = s_0, r_1,..,r_n,r_{n+1},s_{k+1},..,s_1,s_0 = r_0$ be a $t$-cycle of minimal length.
			Let $a_{2i-1} \ T \ a_{2i}$ witness $r_{i-1} \ t \ r_{i}$.
			Let $b_{2i-1} \ T \ b_{2i}$ witness $s_{i_1} \ t \ s_i$.
			Since the $t$-cycle is minimal, the elements are distinct. Therefore the same holds for $a_i,b_j$ (since $\rho$ is a function).
			Consider the $T$-path passsing through $r_0$, $a_1$, $a_2$, $r_1$,..,$a_{2n +2}, r_{n+1} = s_{k+1},..,b_1,s_0$.
			This is a $T$-cycle, contradiction.
		\item $\{\text{classes of } E \} = \{\text{connected components of } t\}$ \\
			Denote with $[x]_t$ the connected component of $x$. It is enough proving that $\forall x \in X \ ([x]_E = [x]_t)$. 
			\begin{itemize}
				\item $[x]_t \subseteq [x]_E$ : \\
					\begin{align*}
						y \in [x]_t & \Rightarrow \exists z_1,..,z_k (x \ t \ z_1 \ t .. t \ z_k \ t \ y) \\
							& \Rightarrow x \ E \ z_1 \ E .. E \ z_k \ E \ y \\
							& \Rightarrow x E y
					\end{align*}

				\item $[x]_E \subseteq [x]_t$ : \\
					\begin{align*}
						y \in [x]_E & \Rightarrow y E x  \\
							& \Rightarrow y F x      \\
							& \Rightarrow \exists z_1,..,z_k (y =: z_0 \ T \ z_1 \ T .. T \ z_k \ T z_{k+1} := x) \\
							& \Rightarrow \bigwedge_{i=0}^{k} \bigg(\rho(z_i,x) = 
									\rho(z_{i+1},x) \lor \rho(z_i,x) \ t \ \rho(z_{i+1},x) \bigg)
					\end{align*}

					At this point, deleting the duplicates (if any) in 
					$\langle y = \rho(z_0,x),.., \rho(z_{k+1},x) = x \rangle $,
					one obtains the desired $t$-sequence witnessing $y \in [x]_E$.
			\end{itemize}
	\end{itemize}

\end{proof}





\begin{esercizio}
	Let $X,Y$ be standard Borel spaces, $a:G \curvearrowright X , b:G \curvearrowright Y $ be Borel actions on $X,Y$ respectively.
	Suppose that $f:X \to Y$ be orbit-wise injective, $G$ equivariant, Borel map.
	Prove that if $E_b$ has the cocycle property, then so does $E_a$.
\end{esercizio}

\begin{proof}
	
	Let $\rho: E_b \to G$ be a cocycle.
	Consider $\delta : E_a \to G, \ (x_1,x_2) \mapsto \rho(f(x_1),f(x_2))$.
	We need to prove the following claims.
	\begin{itemize}
		\item $\delta$ is well defined: \\
			We need to verify that
			$\forall (x_1,x_2) \in E_a \ \bigg( (f(x_1),f(x_2)) \in E_b \bigg)$.
			To see this observe that
			$(x_1,x_2) \in E_a \Rightarrow x_1 = g \cdot x_2 \Rightarrow f(x_1) = f(g \cdot x_2) = g \cdot f(x_2) 
			\Rightarrow f(x_1) E_b f(x_2)$
		\item $\delta$ is cocycle for $E_a$: \\
			We need to verify the following:
			\begin{itemize}
				\item $\delta(x_1,x_3) = \delta(x_2,x_3) \cdot \delta(x_1,x_2)$: \\
					$\delta(x_2,x_3) \cdot \delta(x_1,x_2) =
					\rho(f(x_2),f(x_3)) \cdot \rho(f(x_1),f(x_2)) =
					\rho (f(x_1),f(x_3)) = \delta(x_1,x_3)$
				\item $\delta(x_1,x_2) \cdot x_1 = x_2$: \\
					First observe that $x_1 E_a x_2 \Rightarrow \delta(x_1,x_2) \cdot x_1 E_a x_2$.
					Now, by orbit-wise injectivity of $f$ we just need to check that 
					$f(\delta(x_1,x_2) \cdot x_1) = f(x_2)$.
					This is true because 
					$f(\delta(x_1,x_2) \cdot x_1) = \delta(x_1,x_2) \cdot f(x_1) =
					\rho(f(x_1),f(x_2)) \cdot f(x_1) = f(x_2)$
			\end{itemize}

	\end{itemize}
\end{proof}
